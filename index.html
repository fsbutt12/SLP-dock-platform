// ======================================================== // ENGINE 1: DRIVER IQAMA SCANNER (FORGIVING NAME FILTER) // ======================================================== document.getElementById('iqamaScanner').addEventListener('change', function(e) { const file = e.target.files[0]; if (!file) return; const status = document.getElementById('iqamaStatus'); status.innerHTML = "🌀 Processing Iqama locally..."; status.style.color = "var(--accent-color)"; Tesseract.recognize(file, 'eng').then(({ data: { text } }) => { debugOcr("IQAMA", text); status.innerHTML = "✅ Scan complete!"; status.style.color = "var(--success-color)"; const lines = text.split('\n'); let extractedName = ""; for (let line of lines) { let cleaned = line.trim().toUpperCase(); // Filter out non-name lines common on card formats if (/KINGDOM|SAUDI|ARABIA|MINISTRY|INTERIOR|RESIDENCE|PERMIT|IDENTITY|NUMBER|DATE/i.test(cleaned)) { continue; } // Smarter Match: Capture long sequences of English alphabetical characters, ignoring small misreads const nameMatch = cleaned.match(/\b[A-Z\s]{12,}\b/); if (nameMatch) { extractedName = nameMatch[0].trim(); break; } } if (extractedName) { document.getElementById('driverName').value = extractedName; } else { status.innerHTML = "⚠️ Name line not isolated. Try a closer view."; status.style.color = "var(--accent-color)"; } }).catch(err => { status.innerHTML = "❌ Failed processing card."; status.style.color = "#ef4444"; }); }); // ======================================================== // ENGINE 2: LICENSE PLATE SCANNER (ENGLISH ONLY FILTER) // ======================================================== document.getElementById('plateScanner').addEventListener('change', function(e) { const file = e.target.files[0]; if (!file) return; const status = document.getElementById('plateStatus'); status.innerHTML = "🌀 Isolating plate elements..."; status.style.color = "var(--accent-color)"; Tesseract.recognize(file, 'eng').then(({ data: { text } }) => { debugOcr("PLATE", text); status.innerHTML = "✅ Scan complete!"; status.style.color = "var(--success-color)"; const lines = text.split('\n'); let digits = ""; let letters = ""; lines.forEach(line => { let cleaned = line.replace(/[^A-Z0-9\s]/gi, '').toUpperCase().trim(); // Filter out noisy peripheral words from the image background if (/OPENSOUQ|ELANTRA|HYUNDAI|KSA|SAUDI/i.test(cleaned)) return; // Separate numbers line from English character groups // Checks if line contains the standalone numerical code (like 9192) if (/^\d{3,4}$/.test(cleaned)) { digits = cleaned; } // Checks if line matches the pure English letter plate sequence (3 capitalized characters, e.g., DRB) else if (/^[A-Z]{3}$/.test(cleaned)) { letters = cleaned; } }); // If standalone structural parsing missed it, pull via broad fallback keyword array if (!digits || !letters) { const words = text.split(/\s+/); words.forEach(word => { let cleanWord = word.replace(/[^A-Z0-9]/gi, '').toUpperCase(); if (/^\d{4}$/.test(cleanWord)) digits = cleanWord; if (/^[A-Z]{3}$/.test(cleanWord) && !/KSA/i.test(cleanWord)) letters = cleanWord; }); } if (digits || letters) { document.getElementById('plateNumber').value = `${digits} ${letters}`.trim(); } else { status.innerHTML = "⚠️ Bumper markers blurred. Re-snap."; status.style.color = "var(--accent-color)"; } }).catch(() => { status.innerHTML = "❌ Error processing layout."; status.style.color = "#ef4444"; }); }); // ======================================================== // ENGINE 3: STAFF ID BADGE SCANNER (REMOVES # SYMBOLS) // ======================================================== document.getElementById('officerScanner').addEventListener('change', function(e) { const file = e.target.files[0]; if (!file) return; const status = document.getElementById('officerStatus'); status.innerHTML = "🌀 Locating ID code..."; status.style.color = "var(--accent-color)"; Tesseract.recognize(file, 'eng').then(({ data: { text } }) => { debugOcr("OFFICER BADGE", text); status.innerHTML = "✅ Scan complete!"; status.style.color = "var(--success-color)"; // Match layout structures containing identifier sequences (e.g., LP0022) const badgeMatch = text.match(/LP\d+/i) || text.match(/[A-Z]{2}\d{4}\b/i); if (badgeMatch) { // Force text to drop any remaining hashtag symbols cleanly document.getElementById('securityName').value = badgeMatch[0].replace('#', '').toUpperCase(); } else { const words = text.split(/\s+/); let fallback = ""; for(let w of words) { let clean = w.replace(/[^A-Z0-9]/gi, ''); // Strips hash natively here if(clean.length >= 5 && /[0-9]/.test(clean)) { fallback = clean.toUpperCase(); break; } } if(fallback) { document.getElementById('securityName').value = fallback; } else { status.innerHTML = "⚠️ Badging strings unread. Hold steady."; status.style.color = "var(--accent-color)"; } } }).catch(() => { status.innerHTML = "❌ System fault scanning card."; status.style.color = "#ef4444"; }); });
